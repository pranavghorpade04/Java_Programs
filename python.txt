# Program to print first N prime numbers

def is_prime(num):
    """Function to check if a number is prime."""
    if num < 2:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

# Input: how many prime numbers to print
n = int(input("Enter how many prime numbers you want to print: "))

count = 0
num = 2  # first prime number

print(f"\nFirst {n} Prime Numbers are:")

while count < n:
    if is_prime(num):
        print(num, end=" ")
        count += 1
    num += 1
